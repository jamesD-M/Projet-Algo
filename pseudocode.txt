//Pseudocode détaillant les étapes du processus

// Classe : Piece
// Représente un pion ou une dame sur le plateau de dames

CLASSE Piece
    PROPRIÉTÉ joueur   // le joueur propriétaire de la pièce ("R" ou "B")
    PROPRIÉTÉ estDame  // indique si la pièce est promue en dame (vrai/faux)

    FONCTION constructeur(joueur, estDame = faux)
        METTRE this.joueur = joueur
        METTRE this.estDame = estDame
    FIN FONCTION

    FONCTION toString()
        SI estDame est vrai
            RETOURNER joueur en MAJUSCULE
            // Exemple : "R" = dame rouge, "B" = dame noire
        SINON
            RETOURNER joueur en minuscule
            // Exemple : "r" = pion rouge, "b" = pion noir
        FIN SI
    FIN FONCTION
FIN CLASSE
